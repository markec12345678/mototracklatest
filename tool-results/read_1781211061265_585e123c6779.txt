   100→        setCurrentStyle(found)

   229→  // Keyboard shortcuts
   230→  useEffect(() => {
   231→    function handleKeyDown(e: KeyboardEvent) {
   232→      // Ctrl+Z / Cmd+Z: Undo
   233→      if (e.key === 'z' && (e.ctrlKey || e.metaKey) && !e.shiftKey) {
   234→        e.preventDefault()
   235→        useUndoStore.getState().undo()
   236→        return
   237→      }
   238→      // Ctrl+Y / Cmd+Y or Ctrl+Shift+Z / Cmd+Shift+Z: Redo
   239→      if (
   240→        (e.key === 'y' && (e.ctrlKey || e.metaKey)) ||
   241→        (e.key === 'z' && (e.ctrlKey || e.metaKey) && e.shiftKey)
   242→      ) {
   243→        e.preventDefault()
   244→        useUndoStore.getState().redo()
   245→        return
   246→      }
   247→      // Ctrl+G / Cmd+G: Open coordinate dialog
   248→      if (e.key === 'g' && (e.ctrlKey || e.metaKey)) {
   249→        e.preventDefault()
   250→        setCoordDialogOpen(true)
   251→        return

   253→
   254→      // Don't trigger shortcuts when typing in inputs
   255→      const target = e.target as HTMLElement

   257→        target.tagName === 'INPUT' ||
   258→        target.tagName === 'TEXTAREA' ||
   259→        target.tagName === 'SELECT' ||
   260→        target.isContentEditable
   261→      ) {
   262→        // Allow Escape from inputs
   263→        if (e.key === 'Escape') {
   264→          target.blur()
   265→        }
   266→        return
   267→      }
   268→
   269→      switch (e.key) {
   270→        case '1':
   271→          setToolMode('navigate')
   272→          break
   273→        case '2':
   274→          setToolMode('mark')
   275→          break
   276→        case '3':
   277→          setToolMode('measure')
   278→          break
   279→        case '4':
   280→          setToolMode('directions')
   281→          break
   282→        case '5':
   283→          setToolMode('draw')
   284→          break
   285→        case '6':
   286→          setToolMode('area')
   287→          break
   288→        case '0': {
   289→          const streets = MAP_STYLES.find((s) => s.id === 'streets')
   290→          if (streets) useMapStore.getState().setCurrentStyle(streets)
   291→          break
   292→        }
   293→        case '7': {
   294→          const satellite = MAP_STYLES.find((s) => s.id === 'satellite')
   295→          if (satellite) useMapStore.getState().setCurrentStyle(satellite)
   296→          break
   297→        }
   298→        case '8': {
   299→          const dark = MAP_STYLES.find((s) => s.id === 'dark')
   300→          if (dark) useMapStore.getState().setCurrentStyle(dark)
   301→          break
   302→        }
   303→        case '9': {
   304→          const currentIdx = MAP_STYLES.findIndex((s) => s.id === useMapStore.getState().currentStyle.id)
   305→          const nextIdx = (currentIdx + 1) % MAP_STYLES.length
   306→          useMapStore.getState().setCurrentStyle(MAP_STYLES[nextIdx])
   307→          break
   308→        }
   309→        case '/':
   310→          e.preventDefault()
   311→          document.getElementById('map-search-input')?.focus()
   312→          break
   313→        case 'b':
   314→        case 'B':
   315→          setSidebarOpen(!useMapStore.getState().sidebarOpen)
   316→          break
   317→        case 'f':
   318→        case 'F':
   319→          toggleFullscreen()
   320→          break
   321→        case 'l':
   322→        case 'L':
   323→          handleLocateMe()
   324→          break
   325→        case 'Escape':
   326→          useMapStore.getState().setSelectedMarker(null)
   327→          break
   328→        case '?':
   329→          setShortcutsOpen(true)
   330→          break
   331→      }
   332→    }
   333→
   334→    window.addEventListener('keydown', handleKeyDown)
   335→    return () => window.removeEventListener('keydown', handleKeyDown)
   336→  }, [setToolMode, setSidebarOpen, toggleFullscreen, handleLocateMe, setShortcutsOpen])

   424→              borderRadius: 1,

   427→          {/* Vertical line */}

   434→              borderRadius: 1,

   437→          {/* Center dot */}
