  1763тЖТ  1763тЖТ// Haversine distance calculation
  1764тЖТ  1764тЖТfunction haversineDistance(
  1765тЖТ  1765тЖТ  lat1: number,
  1766тЖТ  1766тЖТ  lon1: number,
  1767тЖТ  1767тЖТ  lat2: number,
  1768тЖТ  1768тЖТ  lon2: number
  1769тЖТ  1769тЖТ): number {
  1770тЖТ  1770тЖТ  const R = 6371 // km
  1771тЖТ  1771тЖТ  const dLat = ((lat2 - lat1) * Math.PI) / 180
  1772тЖТ  1772тЖТ  const dLon = ((lon2 - lon1) * Math.PI) / 180
  1773тЖТ  1773тЖТ  const a =
  1774тЖТ  1774тЖТ    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
  1775тЖТ  1775тЖТ    Math.cos((lat1 * Math.PI) / 180) *
  1776тЖТ  1776тЖТ      Math.cos((lat2 * Math.PI) / 180) *
  1777тЖТ  1777тЖТ      Math.sin(dLon / 2) *
  1778тЖТ  1778тЖТ      Math.sin(dLon / 2)
  1779тЖТ  1779тЖТ  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
  1780тЖТ  1780тЖТ  return R * c
  1781тЖТ  1781тЖТ}

  1783тЖТ  1783тЖТ// Calculate bearing between two points (in degrees)
  1784тЖТ  1784тЖТfunction calculateBearing(
  1785тЖТ  1785тЖТ  lat1: number,
  1786тЖТ  1786тЖТ  lon1: number,
  1787тЖТ  1787тЖТ  lat2: number,
  1788тЖТ  1788тЖТ  lon2: number

  1790тЖТ  1790тЖТ  const dLon = ((lon2 - lon1) * Math.PI) / 180

  1792тЖТ  1792тЖТ  const x =
  1793тЖТ  1793тЖТ    Math.cos((lat1 * Math.PI) / 180) * Math.sin((lat2 * Math.PI) / 180) -
  1794тЖТ  1794тЖТ    Math.sin((lat1 * Math.PI) / 180) * Math.cos((lat2 * Math.PI) / 180) * Math.cos(dLon)
  1795тЖТ  1795тЖТ  const bearing = (Math.atan2(y, x) * 180) / Math.PI
  1796тЖТ  1796тЖТ  return (bearing + 360) % 360
