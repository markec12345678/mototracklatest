    40тЖТ    40тЖТfunction getLayerCategory(layerId: string): keyof LayerVisibility | null {
    41тЖТ    41тЖТ  const lowerId = layerId.toLowerCase()
    42тЖТ    42тЖТ  for (const [category, patterns] of Object.entries(LAYER_PATTERNS)) {
    43тЖТ    43тЖТ    if (patterns.some((pattern) => lowerId.includes(pattern))) {
    44тЖТ    44тЖТ      return category as keyof LayerVisibility
    45тЖТ    45тЖТ    }
    46тЖТ    46тЖТ  }
    47тЖТ    47тЖТ  return null
    48тЖТ    48тЖТ}
