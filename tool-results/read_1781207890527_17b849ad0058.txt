    41тЖТ    41тЖТfunction getLayerCategory(layerId: string): keyof LayerVisibility | null {
    42тЖТ    42тЖТ  const lowerId = layerId.toLowerCase()
    43тЖТ    43тЖТ  for (const [category, patterns] of Object.entries(LAYER_PATTERNS)) {
    44тЖТ    44тЖТ    if (patterns.some((pattern) => lowerId.includes(pattern))) {
    45тЖТ    45тЖТ      return category as keyof LayerVisibility
    46тЖТ    46тЖТ    }
    47тЖТ    47тЖТ  }
    48тЖТ    48тЖТ  return null
    49тЖТ    49тЖТ}
