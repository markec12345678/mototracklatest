    42тЖТ    42тЖТfunction getLayerCategory(layerId: string): keyof LayerVisibility | null {
    43тЖТ    43тЖТ  const lowerId = layerId.toLowerCase()
    44тЖТ    44тЖТ  for (const [category, patterns] of Object.entries(LAYER_PATTERNS)) {
    45тЖТ    45тЖТ    if (patterns.some((pattern) => lowerId.includes(pattern))) {
    46тЖТ    46тЖТ      return category as keyof LayerVisibility
    47тЖТ    47тЖТ    }
    48тЖТ    48тЖТ  }
    49тЖТ    49тЖТ  return null
    50тЖТ    50тЖТ}
