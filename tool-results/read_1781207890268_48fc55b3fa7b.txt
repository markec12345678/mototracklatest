    61тЖТ    61тЖТfunction mapSymbolToCategory(sym?: string): string {
    62тЖТ    62тЖТ  if (!sym) return 'general'
    63тЖТ    63тЖТ  const lower = sym.toLowerCase()
    64тЖТ    64тЖТ  if (lower.includes('restaurant') || lower.includes('food')) return 'restaurant'
    65тЖТ    65тЖТ  if (lower.includes('hotel') || lower.includes('lodging')) return 'hotel'
    66тЖТ    66тЖТ  if (lower.includes('park')) return 'park'
    67тЖТ    67тЖТ  if (lower.includes('star') || lower.includes('favorite')) return 'favorite'
    68тЖТ    68тЖТ  if (lower.includes('shop') || lower.includes('store')) return 'shop'
    69тЖТ    69тЖТ  if (lower.includes('landmark') || lower.includes('museum')) return 'landmark'
    70тЖТ    70тЖТ  return 'general'
    71тЖТ    71тЖТ}
