    78→      const zoomNum = parseFloat(zoomParam)
    79→      if (!isNaN(zoomNum)) {

    81→      }

    84→      const found = MAP_STYLES.find((s) => s.id === styleParam)
    85→      if (found) {
    86→        setCurrentStyle(found)
    87→      }
    88→    }

    97→        lng: lng.toFixed(5),
    98→        zoom: zoom.toFixed(2),
    99→        style: currentStyle.id,
   100→      })

   103→    return () => clearTimeout(timer)

   109→    const params = new URLSearchParams({
   110→      lat: lat.toFixed(5),
   111→      lng: lng.toFixed(5),
   112→      zoom: zoom.toFixed(2),
   113→      style: currentStyle.id,
   114→    })

   120→      const textArea = document.createElement('textarea')
   121→      textArea.value = shareUrl

   123→      textArea.select()

   125→      document.body.removeChild(textArea)

   194→  // Keyboard shortcuts
   195→  useEffect(() => {
   196→    function handleKeyDown(e: KeyboardEvent) {
   197→      // Don't trigger shortcuts when typing in inputs
   198→      const target = e.target as HTMLElement
   199→      if (
   200→        target.tagName === 'INPUT' ||
   201→        target.tagName === 'TEXTAREA' ||
   202→        target.tagName === 'SELECT' ||
   203→        target.isContentEditable

   214→          setToolMode('navigate')
   215→          break
   216→        case '2':
   217→          setToolMode('mark')
   218→          break
   219→        case '3':
   220→          setToolMode('measure')
   221→          break
   222→        case '4':
   223→          setToolMode('directions')
   224→          break
   225→        case '5':
   226→          setToolMode('draw')
   227→          break
   228→        case '0': {
   229→          const streets = MAP_STYLES.find((s) => s.id === 'streets')
   230→          if (streets) useMapStore.getState().setCurrentStyle(streets)
   231→          break

   233→        case '6': {
   234→          const satellite = MAP_STYLES.find((s) => s.id === 'satellite')
   235→          if (satellite) useMapStore.getState().setCurrentStyle(satellite)
   236→          break
   237→        }
   238→        case '7': {
   239→          const dark = MAP_STYLES.find((s) => s.id === 'dark')
   240→          if (dark) useMapStore.getState().setCurrentStyle(dark)
   241→          break

   243→        case '8': {
   244→          const terrain = MAP_STYLES.find((s) => s.id === 'terrain')
   245→          if (terrain) useMapStore.getState().setCurrentStyle(terrain)
   246→          break
   247→        }
   248→        case '9': {
   249→          const currentIdx = MAP_STYLES.findIndex((s) => s.id === useMapStore.getState().currentStyle.id)
   250→          const nextIdx = (currentIdx + 1) % MAP_STYLES.length
   251→          useMapStore.getState().setCurrentStyle(MAP_STYLES[nextIdx])
   252→          break
   253→        }
   254→        case '/':
   255→          e.preventDefault()
   256→          document.getElementById('map-search-input')?.focus()
   257→          break
   258→        case 'b':
   259→        case 'B':
   260→          setSidebarOpen(!useMapStore.getState().sidebarOpen)
   261→          break
   262→        case 'f':
   263→        case 'F':
   264→          toggleFullscreen()
   265→          break
   266→        case 'l':
   267→        case 'L':
   268→          handleLocateMe()
   269→          break
   270→        case 'Escape':
   271→          useMapStore.getState().setSelectedMarker(null)
   272→          break
   273→        case '?':
   274→          setShortcutsOpen(true)
   275→          break
   276→      }
   277→    }
   278→
   279→    window.addEventListener('keydown', handleKeyDown)
   280→    return () => window.removeEventListener('keydown', handleKeyDown)
