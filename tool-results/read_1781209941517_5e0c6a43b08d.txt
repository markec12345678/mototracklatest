    33→// Layer matching patterns
    34→const LAYER_PATTERNS: Record<keyof LayerVisibility, string[]> = {
    35→  water: ['water'],
    36→  roads: ['road'],
    37→  buildings: ['building'],
    38→  parks: ['park', 'landuse'],
    39→  labels: ['label', 'place'],
    40→}
    41→
    42→function getLayerCategory(layerId: string): keyof LayerVisibility | null {
    43→  const lowerId = layerId.toLowerCase()
    44→  for (const [category, patterns] of Object.entries(LAYER_PATTERNS)) {
    45→    if (patterns.some((pattern) => lowerId.includes(pattern))) {
    46→      return category as keyof LayerVisibility
    47→    }
    48→  }
    49→  return null
    50→}
