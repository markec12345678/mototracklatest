    58тЖТ    58тЖТ  try {

  1842тЖТ  1842тЖТ// Haversine distance calculation
  1843тЖТ  1843тЖТfunction haversineDistance(
  1844тЖТ  1844тЖТ  lat1: number,
  1845тЖТ  1845тЖТ  lon1: number,
  1846тЖТ  1846тЖТ  lat2: number,
  1847тЖТ  1847тЖТ  lon2: number
  1848тЖТ  1848тЖТ): number {
  1849тЖТ  1849тЖТ  const R = 6371 // km
  1850тЖТ  1850тЖТ  const dLat = ((lat2 - lat1) * Math.PI) / 180
  1851тЖТ  1851тЖТ  const dLon = ((lon2 - lon1) * Math.PI) / 180
  1852тЖТ  1852тЖТ  const a =
  1853тЖТ  1853тЖТ    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
  1854тЖТ  1854тЖТ    Math.cos((lat1 * Math.PI) / 180) *
  1855тЖТ  1855тЖТ      Math.cos((lat2 * Math.PI) / 180) *
  1856тЖТ  1856тЖТ      Math.sin(dLon / 2) *
  1857тЖТ  1857тЖТ      Math.sin(dLon / 2)
  1858тЖТ  1858тЖТ  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
  1859тЖТ  1859тЖТ  return R * c
  1860тЖТ  1860тЖТ}

  1862тЖТ  1862тЖТ// Calculate bearing between two points (in degrees)
  1863тЖТ  1863тЖТfunction calculateBearing(
  1864тЖТ  1864тЖТ  lat1: number,
  1865тЖТ  1865тЖТ  lon1: number,
  1866тЖТ  1866тЖТ  lat2: number,
  1867тЖТ  1867тЖТ  lon2: number

  1869тЖТ  1869тЖТ  const dLon = ((lon2 - lon1) * Math.PI) / 180

  1871тЖТ  1871тЖТ  const x =
  1872тЖТ  1872тЖТ    Math.cos((lat1 * Math.PI) / 180) * Math.sin((lat2 * Math.PI) / 180) -
  1873тЖТ  1873тЖТ    Math.sin((lat1 * Math.PI) / 180) * Math.cos((lat2 * Math.PI) / 180) * Math.cos(dLon)
  1874тЖТ  1874тЖТ  const bearing = (Math.atan2(y, x) * 180) / Math.PI
  1875тЖТ  1875тЖТ  return (bearing + 360) % 360
