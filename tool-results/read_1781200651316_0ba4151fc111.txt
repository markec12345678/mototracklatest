   185→        target.tagName === 'INPUT' ||

   199→          setToolMode('navigate')
   200→          break
   201→        case '2':
   202→          setToolMode('mark')
   203→          break
   204→        case '3':
   205→          setToolMode('measure')
   206→          break
   207→        case '4':
   208→          setToolMode('directions')
   209→          break
   210→        case '5':
   211→          setToolMode('draw')
   212→          break

   214→          const streets = MAP_STYLES.find((s) => s.id === 'streets')
   215→          if (streets) useMapStore.getState().setCurrentStyle(streets)
   216→          break

   218→        case '6': {
   219→          const satellite = MAP_STYLES.find((s) => s.id === 'satellite')
   220→          if (satellite) useMapStore.getState().setCurrentStyle(satellite)
   221→          break
   222→        }
   223→        case '7': {
   224→          const dark = MAP_STYLES.find((s) => s.id === 'dark')
   225→          if (dark) useMapStore.getState().setCurrentStyle(dark)
   226→          break
   227→        }
   228→        case '8': {
   229→          const terrain = MAP_STYLES.find((s) => s.id === 'terrain')
   230→          if (terrain) useMapStore.getState().setCurrentStyle(terrain)
   231→          break
   232→        }
   233→        case '9': {
   234→          const currentIdx = MAP_STYLES.findIndex((s) => s.id === useMapStore.getState().currentStyle.id)
   235→          const nextIdx = (currentIdx + 1) % MAP_STYLES.length
   236→          useMapStore.getState().setCurrentStyle(MAP_STYLES[nextIdx])
   237→          break
   238→        }
   239→        case '/':
   240→          e.preventDefault()
   241→          document.getElementById('map-search-input')?.focus()
   242→          break
   243→        case 'b':
   244→        case 'B':
   245→          setSidebarOpen(!useMapStore.getState().sidebarOpen)
   246→          break
   247→        case 'f':
   248→        case 'F':
   249→          toggleFullscreen()

   252→        case 'L':
   253→          handleLocateMe()
   254→          break
   255→        case 'Escape':
   256→          useMapStore.getState().setSelectedMarker(null)
   257→          break
   258→        case '?':
   259→          setShortcutsOpen(true)
   260→          break
   261→      }
   262→    }
