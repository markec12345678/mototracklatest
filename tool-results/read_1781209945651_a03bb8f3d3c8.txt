    63→  // Restore map state from URL params on page load

    65→    const params = new URLSearchParams(window.location.search)
    66→    const lat = params.get('lat')

    82→        }, 500)
    83→      }
    84→    }
    85→    if (zoomParam) {
    86→      const zoomNum = parseFloat(zoomParam)
    87→      if (!isNaN(zoomNum)) {
    88→        setZoom(zoomNum)
    89→      }
    90→    }
    91→    if (styleParam) {
    92→      const found = MAP_STYLES.find((s) => s.id === styleParam)
    93→      if (found) {
    94→        setCurrentStyle(found)
    95→      }
    96→    }
    97→  }, [setCenter, setZoom, setCurrentStyle])
    98→

   112→  }, [center, zoom, currentStyle])
   113→
   114→  // Handle map export/screenshot

   116→    const screenshot = await (window as unknown as Record<string, () => Promise<string | null>>).__mapScreenshot?.()
   117→    if (screenshot) {

   125→    }
   126→  }, [])
   127→
   128→  // Handle share URL
   129→  const handleShare = useCallback(() => {
   130→    const [lng, lat] = center
   131→    const params = new URLSearchParams({
   132→      lat: lat.toFixed(5),
   133→      lng: lng.toFixed(5),
   134→      zoom: zoom.toFixed(2),
   135→      style: currentStyle.id,
   136→    })

   138→    navigator.clipboard.writeText(shareUrl).then(() => {
   139→      toast.success('Share URL copied to clipboard!')
   140→    }).catch(() => {
   141→      // Fallback for older browsers
   142→      const textArea = document.createElement('textarea')
   143→      textArea.value = shareUrl
   144→      document.body.appendChild(textArea)
   145→      textArea.select()
   146→      document.execCommand('copy')
   147→      document.body.removeChild(textArea)
   148→      toast.success('Share URL copied to clipboard!')
   149→    })
   150→  }, [center, zoom, currentStyle])
   151→

   155→    window.addEventListener('map-ready', handleMapReady)

   157→    const fallbackTimer = setTimeout(() => {
   158→      if ((window as unknown as Record<string, unknown>).__mainMap) {
   159→        setMapInitialized(true)
   160→      }
   161→    }, 0)
   162→    return () => {
   163→      window.removeEventListener('map-ready', handleMapReady)
   164→      clearTimeout(fallbackTimer)
   165→    }
   166→  }, [])
   167→
   168→  // Welcome back toast for returning users
   169→  useEffect(() => {
   170→    if (savedLocations.length > 0) {
   171→      const timer = setTimeout(() => {
   172→        toast.success(`Welcome back! You have ${savedLocations.length} saved location${savedLocations.length > 1 ? 's' : ''}`)
   173→      }, 1200)
   174→      return () => clearTimeout(timer)
   175→    }
   176→  }, [savedLocations.length])
   177→

   180→    const timer = setTimeout(() => setShowWelcome(false), 10000)
   181→    return () => clearTimeout(timer)
   182→  }, [])
   183→

   190→      document.exitFullscreen()

   193→  }, [])

   197→      navigator.geolocation.getCurrentPosition(
   198→        (position) => {
   199→          const { longitude, latitude, accuracy } = position.coords

   204→          if (flyTo) {

   207→          useMapStore.getState().setGeolocation({ longitude, latitude, accuracy })
   208→        },
   209→        () => {
   210→          // Geolocation denied
   211→        }
   212→      )
   213→    }
   214→  }, [])
   215→

   232→  // Keyboard shortcuts
   233→  useEffect(() => {
   234→    function handleKeyDown(e: KeyboardEvent) {
   235→      // Ctrl+Z / Cmd+Z: Undo
   236→      if (e.key === 'z' && (e.ctrlKey || e.metaKey) && !e.shiftKey) {
   237→        e.preventDefault()
   238→        useUndoStore.getState().undo()
   239→        return
   240→      }
   241→      // Ctrl+Y / Cmd+Y or Ctrl+Shift+Z / Cmd+Shift+Z: Redo
   242→      if (
   243→        (e.key === 'y' && (e.ctrlKey || e.metaKey)) ||
   244→        (e.key === 'z' && (e.ctrlKey || e.metaKey) && e.shiftKey)
   245→      ) {
   246→        e.preventDefault()
   247→        useUndoStore.getState().redo()
   248→        return
   249→      }
   250→      // Ctrl+G / Cmd+G: Open coordinate dialog
   251→      if (e.key === 'g' && (e.ctrlKey || e.metaKey)) {
   252→        e.preventDefault()
   253→        setCoordDialogOpen(true)
   254→        return
   255→      }
   256→
   257→      // Don't trigger shortcuts when typing in inputs
   258→      const target = e.target as HTMLElement
   259→      if (
   260→        target.tagName === 'INPUT' ||
   261→        target.tagName === 'TEXTAREA' ||
   262→        target.tagName === 'SELECT' ||
   263→        target.isContentEditable
   264→      ) {
   265→        // Allow Escape from inputs
   266→        if (e.key === 'Escape') {
   267→          target.blur()
   268→        }
   269→        return
   270→      }
   271→
   272→      switch (e.key) {
   273→        case '1':
   274→          setToolMode('navigate')
   275→          break
   276→        case '2':
   277→          setToolMode('mark')
   278→          break
   279→        case '3':
   280→          setToolMode('measure')
   281→          break
   282→        case '4':
   283→          setToolMode('directions')
   284→          break
   285→        case '5':
   286→          setToolMode('draw')
   287→          break
   288→        case '6':
   289→          setToolMode('area')
   290→          break
   291→        case '0': {
   292→          const streets = MAP_STYLES.find((s) => s.id === 'streets')
   293→          if (streets) useMapStore.getState().setCurrentStyle(streets)
   294→          break
   295→        }
   296→        case '7': {
   297→          const satellite = MAP_STYLES.find((s) => s.id === 'satellite')
   298→          if (satellite) useMapStore.getState().setCurrentStyle(satellite)
   299→          break
   300→        }
   301→        case '8': {
   302→          const dark = MAP_STYLES.find((s) => s.id === 'dark')
   303→          if (dark) useMapStore.getState().setCurrentStyle(dark)
   304→          break
   305→        }
   306→        case '9': {
   307→          const currentIdx = MAP_STYLES.findIndex((s) => s.id === useMapStore.getState().currentStyle.id)
   308→          const nextIdx = (currentIdx + 1) % MAP_STYLES.length
   309→          useMapStore.getState().setCurrentStyle(MAP_STYLES[nextIdx])
   310→          break
   311→        }
   312→        case '/':
   313→          e.preventDefault()
   314→          document.getElementById('map-search-input')?.focus()
   315→          break
   316→        case 'b':
   317→        case 'B':
   318→          setSidebarOpen(!useMapStore.getState().sidebarOpen)
   319→          break
   320→        case 'f':
   321→        case 'F':
   322→          toggleFullscreen()
   323→          break
   324→        case 'l':
   325→        case 'L':
   326→          handleLocateMe()
   327→          break
   328→        case 'Escape':
   329→          useMapStore.getState().setSelectedMarker(null)
   330→          break
   331→        case '?':
   332→          setShortcutsOpen(true)
   333→          break
   334→      }
   335→    }
   336→
   337→    window.addEventListener('keydown', handleKeyDown)
   338→    return () => window.removeEventListener('keydown', handleKeyDown)
   339→  }, [setToolMode, setSidebarOpen, toggleFullscreen, handleLocateMe, setShortcutsOpen])

   391→              </div>

   396→                  <div className="h-1.5 w-1.5 rounded-full bg-emerald-500 animate-bounce" style={{ animationDelay: '150ms' }} />
   397→                  <div className="h-1.5 w-1.5 rounded-full bg-emerald-500 animate-bounce" style={{ animationDelay: '300ms' }} />

   416→        <div className="absolute inset-0 pointer-events-none z-[15]" aria-hidden="true">

   418→          <div
   419→            className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2"
   420→            style={{
   421→              width: 60,

   423→              background: toolMode === 'mark' ? 'rgba(239,68,68,0.6)' : toolMode === 'measure' ? 'rgba(245,158,11,0.6)' : toolMode === 'draw' ? 'rgba(34,197,94,0.6)' : toolMode === 'area' ? 'rgba(139,92,246,0.6)' : 'rgba(6,182,212,0.6)',
   424→              borderRadius: 1,
   425→            }}
   426→          />
   427→          {/* Vertical line */}
   428→          <div
   429→            className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2"
   430→            style={{
   431→              width: 2,
   432→              height: 60,
   433→              background: toolMode === 'mark' ? 'rgba(239,68,68,0.6)' : toolMode === 'measure' ? 'rgba(245,158,11,0.6)' : toolMode === 'draw' ? 'rgba(34,197,94,0.6)' : toolMode === 'area' ? 'rgba(139,92,246,0.6)' : 'rgba(6,182,212,0.6)',
   434→              borderRadius: 1,
   435→            }}
   436→          />
   437→          {/* Center dot */}
   438→          <div
   439→            className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 rounded-full"

   448→
   449→      {/* Sidebar - responsive */}

   461→      >
   462→        {/* Responsive padding for desktop sidebar */}

   477→            onClick={() => {

   527→            onClick={() => setShortcutsOpen(true)}
   528→            title="Keyboard Shortcuts"
